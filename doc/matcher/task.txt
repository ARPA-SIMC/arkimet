--- Matching ODIMH5 task

-- ODIMH5
-- Syntax: task:value
-- Matches if values is found inside the metadata
-- if value is empty, any data with task attribute is matched

sample:set(arki_origin.task("prova 123"))

ensure_matches(sample, "task:")
ensure_matches(sample, "task:prova")
ensure_matches(sample, "task:123")

ensure_not_matches(sample, "task:aaaaaaaaaaaaaaaa")
ensure_not_matches(sample, "task:prova123")
